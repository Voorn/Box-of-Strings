# Naturality of swap (Comulti) #

e[2]{
	0 |x1: 2 -> 2; 
	0 |a2: 1 -> 2;
}={
	1 |a2: 1 -> 2;
	0 |x1: 2 -> 2; 
	1 |x1: 2 -> 2; 
}

e[2]{
	0 |x1: 2 -> 2; 
	1 |a2: 1 -> 2;
}={
	0 |a2: 1 -> 2;
	1 |x1: 2 -> 2; 
	0 |x1: 2 -> 2; 
}

# Naturality of swap (co-unit) #

e[2]{
	0 |x1: 2 -> 2; 
	0 |a2: 1 -> 0;
}={
	1 |a2: 1 -> 0;
}

e[2]{
	0 |x1: 2 -> 2; 
	1 |a2: 1 -> 0;
}={
	0 |a2: 1 -> 0;
}

# Naturality of swap (multi) #

e[3]{
	0 |a2: 2 -> 1; 
	0 |x1: 2 -> 2;
}={
	1 |x1: 2 -> 2;
	0 |x1: 2 -> 2;
	1 |a2: 2 -> 1;
}

e[3]{
	1 |a2: 2 -> 1; 
	0 |x1: 2 -> 2;
}={
	0 |x1: 2 -> 2;
	1 |x1: 2 -> 2;
	0 |a2: 2 -> 1;
}
