# Show-case of leaky categories: symmetric encryption + notion of leaked variable "l" #

Import{crypto}

# Since leaks cannot be discarded, we limit naturality of discard #
Schema{naturality}{x.d}{x.s.c.d.?}

# Leaks addition #
Operation{l}{}{1}{1}

Schema{naturality}{x.c}{x.s.c.d.l}

# Leaks can be copied #

# LEAK #

Define{2}{
	0.l;
	0.x;

	0.x;
	1.l;
}

	0.x;
	0.l;
