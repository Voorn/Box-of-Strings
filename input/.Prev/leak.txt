}={
	0 |x1: 2 -> 2;
	1 |l: 1 -> 1;
}

	1 |l: 1 -> 1;
	0 |x1: 2 -> 2;
}={
	0 |x1: 2 -> 2;
	0 |l: 1 -> 1;
}

}

	0 |l: 1 -> 1;
	0 |c2: 1 -> 2;

	0 |c2: 1 -> 2;
	1 |l: 1 -> 1;
}

# Symmetric Key Decrypt# 
Lemma[2]{
	0 |c2: 1 -> 2;
	1 |s: 2 -> 1;
	1 |l: 1 -> 1;
	0 |s: 2 -> 1;
}={
	1 |c2: 1 -> 2;
	0 |s: 2 -> 1;
	0 |l: 1 -> 1;
	0 |d2: 1 -> 0;
}

Assume[2]{
	1 |c2: 1 -> 2;
	0 |z: 2 -> 1;
	0 |l: 1 -> 1;
	0 |z: 2 -> 1;
}={
	0 |c2: 1 -> 2;
	1 |z: 2 -> 1;
	1 |l: 1 -> 1;
	1 |d2: 1 -> 0;
}

Define[0]{
	0 |?: 0 -> 1;
	0 |l: 1 -> 1;

}={}

Example[1]{
	0 |?: 0 -> 1;
	0 |c2: 1 -> 2;
	1 |s: 2 -> 1;
	1 |l: 1 -> 1;
	0 |s: 2 -> 1;
}={}


Define[2]{
	0 |l: 1 -> 1;
	1 |l: 1 -> 1;
	0 |s: 2 -> 1;
}={
	0 |l: 1 -> 1;
	0 |s: 2 -> 1;
	0 |l: 1 -> 1;
}={
	1 |l: 1 -> 1;
	0 |s: 2 -> 1;
	0 |l: 1 -> 1;
}

Example[1]{

	0 |l: 1 -> 1;
