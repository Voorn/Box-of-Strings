# Symmetric Key Decrypt# 
Lemma[2]{
	0 |c2: 1 -> 2;
	1 |s: 2 -> 1;
	1 |l: 1 -> 1;
	0 |s: 2 -> 1;
}={
	1 |c2: 1 -> 2;
	0 |s: 2 -> 1;
	0 |l: 1 -> 1;
	0 |d2: 1 -> 0;
}

	1 |c2: 1 -> 2;

	0 |s: 2 -> 1;

	1 |s: 2 -> 1;
	1 |l: 1 -> 1;
	1 |d2: 1 -> 0;
}
